ســكـن الـجـنـونُ أنَـامـــلـي ويــراعـي
ومــضــى يُــدنِّــس لَــوعــة الـمُـلـتاع
وجــرت عـجـائب ما ظـنـنت حُـدوثـها
فـــي مـنـتهى الـحـسرات و الأوجــاع
صَـمْتي يُـحشرج فـوق لَــحْد خطيئتي
ويــــزفّ أنــفـاس الـــرّدى لـلــسّـاعي
حـتّـى الـحـجارة قـد بـكـت و تـفـتّتت
مـن هـول ما صـنعت سِـنين ضياعي
شَهقت على ظمئي هـــواجس حيرتي
وَهَــوَتْ عـلـى طـلل الـرّؤى أطـمـاعي
فــأنـاخَ لَــفْـحُ الـتّـيه وَجْــهَ حـقـيقتي
فــــي ســاحـة الــخــذلان دون قــنـاع
لو يـمـتطي الـنّسيان ظـهر فـجيعتي
ويَـزجّـها فــي الـغـيب قـبل سـماعي
أحـبـبـتها والـنّـفـس تَرشـف لَـعْـنتي
بــئــس الــغــرام إذا دعــا لــوداعـي
مـا كـنـت أحـسـبها سـتـنقض غـزلها
وتــلـوك عــرضـي بـالـجوى الـخـدّاع !
حـــلّ الــجـفـاف بـأرضـهـا وسـمـائـها
وتــــلا الأنِـــيـــنُ فَــجَــائِـعَ الـــــزّرّاع
هـجـرت عـنـادل شـوقـها مـن نـبضها
فَـسَـلَخْتُ مَــا خَــطَّ الـهَــوَى بـذراعي
يـا قـلب قَـبْـل الـقِـيلِ حُـبّـك قِـبْـلَتِي
مـاذا جـرى ؟... أمْ قَـلّ حــظّ الرّاعي !!
قـد أقـتـفـي أثـــر الـخـيال بـلـهفتي
مـن بـعد عـصف الـحبّ فـي الأصـقـاع
أحـبـبتها والـشّـكّ غـرغـر فـي دمـي
غـــاب الـحِـجـا فـــي لـحـظة الإمـتـاع
يـغـتـالني صبـري بِـضـحـكة غـدرهـا
ويــزيـد قـصـف الـلّـمز وقع صـداعـي
مـن لَــمْ يَــرَ الـمـرآة تـلـبس شـكـله
فــي ســورة الإنـسـان قـــل يـا نـاعي
