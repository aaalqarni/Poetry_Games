كلُّ شيءٍ بقضاءٍ وقَدَر
هكذا المعلومُ
والذي يقضي به حكم النظر
سرُّه مكتوم
كلُّ من أشهده سرّ القدر
ربُّه يعلمُ
إنَّ بالحكم الذي منه ظهر
عينه يحكمُ
عجباً قيمن له نعتُ البشر
وهو لا يفهمُ
والذي يشهدُه نورُ القمر
فهو المرحومُ
والذي عُيِّب عنه واستسرَّ
ذلك المحرومُ
شاهد النقل الذي حيرني
وبه أحيى
ودليلُ العقلِ قد صيَّرني
مُنكِراً أشْيا
فتراني عندما خيّرني
أكره المحيا
فأنا ما بين عقلٍ وخبر
ظالمٌ وظلومْ
فإذا سُرِّحتُ من سجنِ الفكر
قمتُ بالقيوم
بالتجلِّي في التدلِّي قلتُ به
فأبى عقلي
والتجلي في التجلِّي منه به
قال لي قل لي
انت مني عينُ ظلِّي فانتبه
بالهوى من لي
إن جرى الأمرُ على حكمِ البَصَر
قلت بالمفهومْ
أو جرى الأمر على حكم العِبَره
ينتفي المرسوم
لو أنَّ ما بي من شؤون العبادِ
وكلُّ ما يجري
يكون بالسبعِ الطباقِ الشدادِ
يسكّن عن دورِ
إنَّ الذي كانَ مسبيّ مراد
لصاحبِ الأمر
الصبر أولى بي من أجل الظفر
وإنه موهومُ
فاشرب رحيقاً عند وقتِ السَّحَر
مِزاجُه تسنيم
بساحلِ البحر رأيتُ التي
ما زلتُ ألغيها
فقلتُللنفسِ ترى قبلتي
بالله أبغيها
فأنشدتْ تخبر عن جملتي
وذاك يطيعها
ليتني رملٌ على شطِّ البحر
يا ابني أو أطومْ
وترى عيني مذ تطلعُ سِحر
لبلادِ الرُّوم
