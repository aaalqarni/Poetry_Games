الحبيب عرفتو
وأنا منه خايف
ما يحبك إِلا
من هو بيك عارف
مُذ عَرفت ربِّي
زالت عن الأغيار
وانْشرحْ لِي قَلْبي
وبدَت لِي أسْرار
وأنا طول حَيَاتي
في نُور وأنْوار
طُول حَياتِي نَبْقى
في سر الوَظّائف
ما يحبكْ إِلاَّ
من هو بِيك عارِف
يا فقِير تجرَّدْ
عن ثوب البطاله
واتبع الحقائق
وقُل كَيْف قاله
واستمسك يا عارف
بِحبْل الوصاله
ولتكُن لنفسك
عاصي ومخَالف
ما يحبك إِلاَّ
مَن هُوَ بيك عارف
أترك الخلايق
يا بطال واجهد
واقطع العلائق
وأجود وازهد
يبصر الحقائق
نُور قلبك ويشهد
ويُوريك حبيبك
من صنع اللَّطائف
ما يحبك إِلاَّ
من هُوَ بيك عارف
يا فقير تخلا
عن هوى الخليقه
واستمسك يا عارف
بأهل الطريقه
وتكون تتبع
لأهْل الحقِيقه
قريب أنت منهم
كيس وملاطف
ما يحبك إلاَّ
من هو بِيك عارِف
