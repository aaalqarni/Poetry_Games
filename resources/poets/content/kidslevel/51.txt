ما أجملَ القُرآنْ
يعلِّمُ الإنسانْ
فضائلَ الإيمانْ
والخيرَ والإحسانْ
ما أجملَ القرآنْ
إذْ جاءَ بِالبيانْ
حُروفُهُ الحِسَانْ
كالدُّرِّ والجُمانْ
تُثبِّتُ الجَنانْ
وتُبْرِئُ الأبدانْ
وتُبعِدُ النسيانْ
وتَدفَعُ الأحزانْ
بالعِلم وَالإيمانْ
والحقِّ والبرهانْ
تبقَى مَدى الزمانْ
مَكارمُ القُرآنْ
تبقَى مَدى الزمانْ
صحائفُ البَيانْ
